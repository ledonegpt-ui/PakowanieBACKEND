   38:         $daysBack = env_int_val('SUB_DOC_DAYS_BACK', (int)($this->cfg['subiekt']['days_back'] ?? 1));

   40:         if ($daysBack > 30) $daysBack = 30;
   41: 
   42:         $today = new \DateTimeImmutable('today');
   43:         $from  = $today->modify('-' . $daysBack . ' day')->setTime(0, 0, 0);
   44:         $to    = $today->setTime(23, 59, 59);

   46:         $pattern = (string)env_val('SUB_PODTYTUL_PATTERN', (string)($this->cfg['subiekt']['podtytul_pattern'] ?? '*'));
   47:         if ($pattern === '') $pattern = '*';
   48: 
   49:         $types = $this->parseIntList((string)env_val('SUB_DOC_TYPES', (string)($this->cfg['subiekt']['doc_types'] ?? '')));
   50: 
   51:         $magIdStr = trim((string)env_val('SUB_MAG_ID', (string)($this->cfg['subiekt']['mag_id'] ?? '')));
   52:         $magId = ($magIdStr !== '' && ctype_digit($magIdStr)) ? (int)$magIdStr : null;
   53: 
   54:         $select = [
   55:             "d.dok_Id AS dok_Id",
   56:             "d.{$s['noCol']} AS dok_NrPelny",
   57:             "d.{$s['dateCol']} AS dok_Data",
   58:             "d.{$s['podCol']} AS dok_Podtytul",
   59:         ];
   60:         if (!empty($s['uwCol']))  $select[] = "d.{$s['uwCol']} AS dok_Uwagi";

   63:         $sql = "SELECT " . implode(", ", $select) . "
   64:                 FROM {$s['docTable']} d

   66:                   AND d.{$s['podCol']} LIKE :pt";
   67: 
   68:         $bind = [
   69:             ':from'  => $from->format('Y-m-d H:i:s'),
   70:             ':to'    => $to->format('Y-m-d H:i:s'),
   71:             ':pt'    => '%' . $pattern . '%',
   72:         ];
   73: 
   74:         if ($types && !empty($s['typCol'])) {
   75:             $in = [];
   76:             $i = 0;
   77:             foreach ($types as $t) {
   78:                 $ph = ':t' . $i++;
   79:                 $in[] = $ph;
   80:                 $bind[$ph] = $t;
   81:             }
   82:             $sql .= " AND d.{$s['typCol']} IN (" . implode(',', $in) . ")";
   83:         }

   86:             $sql .= " AND d.{$s['magDocCol']} = :magD";

   88:         }

   92:         $st = $this->mssql->prepare($sql);

   97:     /**
   98:      * Zwraca mapę: docId => items[]
   99:      * (wewnętrznie chunkuje IN(), żeby nie zabić limitów parametrów MSSQL)
  100:      *
  101:      * @param array<int,int> $docIds
  102:      * @return array<int,array<int,array<string,mixed>>>
  103:      */
  104:     public function fetchItemsForDocs(array $docIds): array
  105:     {
  106:         $docIds = array_values(array_unique(array_filter($docIds, function($v){
  107:             return is_int($v) && $v > 0;
  108:         })));
  109:         if (!$docIds) return [];
  110: 
  111:         $s = $this->getSchema();
  112: 
  113:         $magIdStr = trim((string)env_val('SUB_MAG_ID', (string)($this->cfg['subiekt']['mag_id'] ?? '')));
  114:         $magId = ($magIdStr !== '' && ctype_digit($magIdStr)) ? (int)$magIdStr : null;
  115: 
  116:         $out = [];
  117: 
  118:         foreach (array_chunk($docIds, 500) as $chunk) {
  119:             $ph = implode(',', array_fill(0, count($chunk), '?'));
  120: 
  121:             $sel = [
  122:                 "p.{$s['posDocCol']} AS dok_Id",
  123:                 "p.{$s['obIdCol']}  AS ob_Id",
  124:                 "p.{$s['towIdCol']} AS ob_TowId",
  125:                 "p.{$s['qtyCol']}   AS ob_Ilosc",
  126:                 ($s['priceCol'] ? "p.{$s['priceCol']}" : "NULL") . " AS ob_CenaBrutto",
  127:                 "t.{$s['twSymbolCol']} AS tw_Symbol",
  128:                 "t.{$s['twNameCol']}   AS tw_Nazwa",
  129:                 ($s['twOpisCol'] ? "t.{$s['twOpisCol']}" : "NULL") . " AS tw_Opis",
  130:             ];
  131: 
  132:             $sql = "SELECT " . implode(", ", $sel) . "
  133:                     FROM {$s['posTable']} p
  134:                     LEFT JOIN {$s['towTable']} t ON p.{$s['towIdCol']} = t.{$s['twIdCol']}
  135:                     WHERE p.{$s['posDocCol']} IN ($ph)";
  136: 
  137:             $bind = $chunk;
  138: 
  139:             if ($magId !== null && !empty($s['magPosCol'])) {
  140:                 $sql .= " AND p.{$s['magPosCol']} = ?";
  141:                 $bind[] = $magId;
  142:             }
  143: 
  144:             $sql .= " ORDER BY p.{$s['posDocCol']} ASC, p.{$s['obIdCol']} ASC";
  145: 
  146:             $st = $this->mssql->prepare($sql);
  147:             $st->execute($bind);
  148:             $rows = $st->fetchAll(\PDO::FETCH_ASSOC) ?: [];
  149: 
  150:             foreach ($rows as $r) {
  151:                 $did = (int)$r['dok_Id'];
  152:                 if (!isset($out[$did])) $out[$did] = [];
  153:                 $out[$did][] = $r;
  154:             }
  155:         }
  156: 
  157:         return $out;
  158:     }

  207:     private function parseIntList(string $csv): array
  208:     {
  209:         if (trim($csv) === '') return [];
  210:         $out = [];
  211:         foreach (explode(',', $csv) as $p) {
  212:             $p = trim($p);
  213:             if ($p === '' || !ctype_digit($p)) continue;
  214:             $out[] = (int)$p;
  215:         }
  216:         return array_values(array_unique($out));
  217:     }

  219:     /** autodetekcja tabel/kolumn + wybór tw_Opis jeśli istnieje */
  220:     private function detectSchema(\DateTimeImmutable $from, \DateTimeImmutable $to, string $pattern, array $types): array
  221:     {
  222:         $docTables = $this->mssql->query("
  223:             SELECT TABLE_NAME
  224:             FROM INFORMATION_SCHEMA.COLUMNS
  225:             WHERE TABLE_NAME LIKE 'dok%'
  226:             GROUP BY TABLE_NAME
  227:             HAVING SUM(CASE WHEN COLUMN_NAME='dok_Id' THEN 1 ELSE 0 END) > 0
  228:                AND SUM(CASE WHEN COLUMN_NAME='dok_Podtytul' THEN 1 ELSE 0 END) > 0

  230:             ORDER BY TABLE_NAME ASC
  231:         ")->fetchAll(\PDO::FETCH_COLUMN);
  232: 
  233:         $posTables = $this->mssql->query("
  234:             SELECT TABLE_NAME
  235:             FROM INFORMATION_SCHEMA.COLUMNS
  236:             WHERE TABLE_NAME LIKE 'dok%'
  237:             GROUP BY TABLE_NAME
  238:             HAVING SUM(CASE WHEN COLUMN_NAME='ob_Id' THEN 1 ELSE 0 END) > 0
  239:                AND SUM(CASE WHEN COLUMN_NAME='ob_TowId' THEN 1 ELSE 0 END) > 0
  240:                AND SUM(CASE WHEN COLUMN_NAME IN ('ob_Ilosc','ob_Il') THEN 1 ELSE 0 END) > 0
  241:                AND SUM(CASE WHEN COLUMN_NAME IN ('ob_DokMagId','ob_DokHanId','ob_DokId','ob_DokumentId') THEN 1 ELSE 0 END) > 0
  242:             ORDER BY TABLE_NAME ASC
  243:         ")->fetchAll(\PDO::FETCH_COLUMN);
  244: 
  245:         if (!$docTables) throw new \RuntimeException('Brak tabeli dokumentów dok_*');
  246:         if (!$posTables) throw new \RuntimeException('Brak tabeli pozycji dok_*');
  247: 
  248:         $towTable = $this->mssqlTableExists('tw__Towar') ? 'tw__Towar' : $this->detectTowTableAny();
  249:         if (!$towTable) throw new \RuntimeException('Brak tabeli towarów (tw__Towar lub podobna)');
  250: 
  251:         $towCols = $this->mssqlColumns($towTable);
  252:         $twIdCol = $this->firstCol($towCols, ['tw_Id']);
  253:         $twSymbolCol = $this->firstCol($towCols, ['tw_Symbol']);
  254:         $twNameCol = $this->firstCol($towCols, ['tw_Nazwa']);
  255: 
  256:         $twOpisCol = $this->firstCol($towCols, ['tw_Opis','tw_OpisWWW','tw_OpisDodatkowy','tw_OpisKrotki','tw_Uwagi','tw_Opis2','tw_Opis3']);
  257: 
  258:         if (!$twIdCol || !$twSymbolCol || !$twNameCol) {
  259:             throw new \RuntimeException("Tabela {$towTable} nie ma tw_Id/tw_Symbol/tw_Nazwa");
  260:         }
  261: 
  262:         foreach ($docTables as $dt) {
  263:             $dt = (string)$dt;
  264:             $docCols = $this->mssqlColumns($dt);
  265:             $dateCol = $this->firstCol($docCols, ['dok_DataWyst','dok_Data','dok_DataWystaw']);
  266:             $noCol   = $this->firstCol($docCols, ['dok_NrPelny','dok_NrPelnyDruk','dok_NrPelnyTekst']);
  267:             $podCol  = $this->firstCol($docCols, ['dok_Podtytul']);
  268:             if (!$dateCol || !$noCol || !$podCol) continue;
  269: 
  270:             $typCol = $this->firstCol($docCols, ['dok_Typ','dok_Podtyp']);
  271:             $uwCol  = $this->firstCol($docCols, ['dok_Uwagi','dok_Opis']);
  272:             $magDocCol = $this->firstCol($docCols, ['dok_MagId','dok_MagazynId','dok_Mag']);
  273: 
  274:             // próbka dokumentu w zakresie
  275:             $bind = [
  276:                 ':from' => $from->format('Y-m-d H:i:s'),
  277:                 ':to'   => $to->format('Y-m-d H:i:s'),
  278:                 ':pt'   => '%' . $pattern . '%',
  279:             ];
  280:             $sql = "SELECT TOP 1 dok_Id FROM {$dt}
  281:                     WHERE {$dateCol} >= :from AND {$dateCol} <= :to
  282:                       AND {$podCol} LIKE :pt";

  284:                 $in = [];

  288:             }
  289:             $sql .= " ORDER BY {$dateCol} DESC";
  290: 
  291:             $st = $this->mssql->prepare($sql);
  292:             $st->execute($bind);
  293:             $sampleDocId = (int)($st->fetchColumn() ?: 0);
  294:             if ($sampleDocId < 1) continue;
  295: 
  296:             foreach ($posTables as $pt) {
  297:                 $pt = (string)$pt;
  298:                 $posCols = $this->mssqlColumns($pt);
  299: 
  300:                 $obIdCol  = $this->firstCol($posCols, ['ob_Id']);

  306:                 $magPosCol = $this->firstCol($posCols, ['ob_MagId','ob_MagazynId','ob_Mag']);

  315:                     $chk->execute([':id' => $sampleDocId]);
  316:                     $one = $chk->fetchColumn();

  318:                         return [
  319:                             'docTable' => $dt,
  320:                             'posTable' => $pt,
  321:                             'towTable' => $towTable,
  322:                             'dateCol' => $dateCol,
  323:                             'noCol' => $noCol,
  324:                             'podCol' => $podCol,
  325:                             'typCol' => $typCol,

  354:     private function detectTowTableAny(): ?string
  355:     {
  356:         $st = $this->mssql->query("
  357:             SELECT TOP 50 TABLE_NAME
  358:             FROM INFORMATION_SCHEMA.COLUMNS
  359:             WHERE TABLE_NAME LIKE 'tw%'
  360:             GROUP BY TABLE_NAME
  361:             HAVING SUM(CASE WHEN COLUMN_NAME='tw_Id' THEN 1 ELSE 0 END) > 0
  362:                AND SUM(CASE WHEN COLUMN_NAME='tw_Symbol' THEN 1 ELSE 0 END) > 0
  363:                AND SUM(CASE WHEN COLUMN_NAME='tw_Nazwa' THEN 1 ELSE 0 END) > 0
  364:             ORDER BY TABLE_NAME ASC
  365:         ");
  366:         $t = $st ? $st->fetchColumn() : false;
  367:         return $t ? (string)$t : null;

  370:     private function mssqlColumns(string $table): array
  371:     {
  372:         if (isset($this->mssqlColsCache[$table])) return $this->mssqlColsCache[$table];
  373: 
  374:         $st = $this->mssql->prepare("SELECT COLUMN_NAME FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME=:t");
  375:         $st->execute([':t'=>$table]);
  376: 
  377:         $cols = [];
  378:         foreach ($st->fetchAll(\PDO::FETCH_ASSOC) as $r) $cols[] = (string)$r['COLUMN_NAME'];
  379:         $this->mssqlColsCache[$table] = $cols;
  380:         return $cols;
  381:     }

  383:     private function firstCol(array $cols, array $candidates): ?string
  384:     {
  385:         foreach ($candidates as $c) if (in_array($c, $cols, true)) return $c;
  386:         return null;
  387:     }
