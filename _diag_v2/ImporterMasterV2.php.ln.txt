  189:     private function extractOrderCode(string $podtytul): ?string
  190:     {
  191:         if ($podtytul === '') return null;

  194:         if ($code === '' || strlen($code) > 32) return null;
  195:         if (!preg_match('/^[0-9A-Z]+$/', $code)) return null;
  196:         return $code;
  197:     }

  199:     private function acquireLock(string $name, int $timeoutSec): bool
